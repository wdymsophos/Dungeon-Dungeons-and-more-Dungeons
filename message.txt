void printl(const char *texto, int atraso_ms)
{
    for (int i = 0; texto[i] != '\0'; i++)
    {
        printf("%c", texto[i]); // ? printf ao invés de printl
        fflush(stdout); // Força impressão imediata
        Sleep(atraso_ms); // Pausa em milissegundos
    }
    printf("\n"); // Nova linha ao final
}
